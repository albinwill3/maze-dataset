 34def tokens_between(
 35    tokens: list[str],
 36    start_value: str,
 37    end_value: str,
 38    include_start: bool = False,
 39    include_end: bool = False,
 40    except_when_tokens_not_unique: bool = False,
 41) -> list[str]:
 42    if start_value == end_value:
 43        raise ValueError(
 44            f"start_value and end_value cannot be the same: {start_value = } {end_value = }"
 45        )
 46    if except_when_tokens_not_unique:
 47        if (tokens.count(start_value) != 1) or (tokens.count(end_value) != 1):
 48            raise ValueError(
 49                "start_value or end_value is not unique in the input tokens:",
 50                f"{tokens.count(start_value) = } {tokens.count(end_value) = }"
 51                f"{start_value = } {end_value = }",
 52                f"{tokens = }",
 53            )
 54    else:
 55        if (tokens.count(start_value) < 1) or (tokens.count(end_value) < 1):
 56            raise ValueError(
 57                "start_value or end_value is not present in the input tokens:",
 58                f"{tokens.count(start_value) = } {tokens.count(end_value) = }",
 59                f"{start_value = } {end_value = }",
 60                f"{tokens = }",
 61            )
 62
 63    start_idx: int = tokens.index(start_value) + int(not include_start)
 64    end_idx: int = tokens.index(end_value) + int(include_end)
 65
 66    assert start_idx < end_idx, "Start must come before end"
 67
 68    return tokens[start_idx:end_idx]

 77def get_path_tokens(tokens: list[str], trim_end: bool = False) -> list[str]:
 78    """The path is considered everything from the first path coord to the path_end token, if it exists."""
 79    if SPECIAL_TOKENS.PATH_START not in tokens:
 80        raise ValueError(
 81            f"Path start token {SPECIAL_TOKENS.PATH_START} not found in tokens:\n{tokens}"
 82        )
 83    start_idx: int = tokens.index(SPECIAL_TOKENS.PATH_START) + int(trim_end)
 84    end_idx: int | None = None
 85    if trim_end and (SPECIAL_TOKENS.PATH_END in tokens):
 86        end_idx = tokens.index(SPECIAL_TOKENS.PATH_END)
 87    return tokens[start_idx:end_idx]

125def get_relative_direction(coords: Int[np.ndarray, "prev_cur_next=3 row_col=2"]) -> str:
126    """Returns the relative first-person direction token corresponding to traveling from `coords[1]` to `coords[2]`.
127    # Parameters
128    - `coords`: Contains 3 Coords, each of which must neighbor the previous Coord.
129      - `coords[0]`: The previous location, used to determine the current absolute direction that the "agent" is facing.
130      - `coords[1]`: The current location
131      - `coords[2]`: The next location. May be equal to the current location.
132    """
133    if coords.shape != (3, 2):
134        raise ValueError(f"`coords` must have shape (3,2). Got {coords.shape} instead.")
135    directions = coords[1:] - coords[:-1]
136    if not np.all(np.linalg.norm(directions, axis=1) <= np.array([1.1, 1.1])):
137        # Use floats as constant since `np.linalg.norm` returns float array
138        raise ValueError(
139            f"Adjacent `coords` must be neighboring or equivalent. Got {coords} instead."
140        )
141    if np.array_equal(coords[1], coords[2]):
142        return VOCAB.PATH_STAY
143    if np.array_equal(coords[0], coords[2]):
144        return VOCAB.PATH_BACKWARD
145    if np.array_equal(coords[0], coords[1]):
146        raise ValueError(
147            f"Previous first-person direction indeterminate from {coords=}."
148        )
149    if np.array_equal(directions[0], directions[1]):
150        return VOCAB.PATH_FORWARD
151    directions = np.append(
152        directions, [[0], [0]], axis=1
153    )  # Augment to represent unit basis vectors in 3D
154    match np.cross(directions[0], directions[1])[-1]:
155        case 1:
156            return VOCAB.PATH_LEFT
157        case -1:
158            return VOCAB.PATH_RIGHT

167def str_is_coord(coord_str: str, allow_whitespace: bool = True) -> bool:
168    """return True if the string represents a coordinate, False otherwise"""
169
170    warnings.warn(
171        "`util.str_is_coord` only supports legacy UT strings. Function will be replaced with a generalized version in a future release.",
172        TokenizerPendingDeprecationWarning,
173    )
174    strip_func: Callable[[str], str] = lambda x: x.strip() if allow_whitespace else x
175
176    coord_str = strip_func(coord_str)
177
178    return all(
179        [
180            coord_str.startswith("("),
181            coord_str.endswith(")"),
182            "," in coord_str,
183            all(
184                [
185                    strip_func(x).isdigit()
186                    for x in strip_func(coord_str.lstrip("(").rstrip(")")).split(",")
187                ]
188            ),
189        ]
190    )

203def _coord_to_strings_UT(coord: typing.Sequence[int]) -> list[str]:
204    """convert a coordinate to a string: `(i,j)`->"(i,j)"

218def coord_str_to_tuple(
219    coord_str: str, allow_whitespace: bool = True
220) -> tuple[int, ...]:
221    """convert a coordinate string to a tuple"""
222    strip_func: Callable[[str], str] = lambda x: x.strip() if allow_whitespace else x
223    coord_str = strip_func(coord_str)
224    stripped: str = strip_func(coord_str.lstrip("(").rstrip(")"))
225    return tuple(int(strip_func(x)) for x in stripped.split(","))

240def coords_string_split_UT(coords: str) -> list[str]:
241    """Splits a string of tokens into a list containing the UT tokens for each coordinate.
242
243    Not capable of producing indexed tokens ("(", "1", ",", "2", ")"), only unique tokens ("(1,2)").
244    Non-whitespace portions of the input string not matched are preserved in the same list:
245    "(1,2) <SPECIAL_TOKEN> (5,6)" -> ["(1,2)", "<SPECIAL_TOKEN>", "(5,6)"]
246    """
247    # ty gpt4
248    return re.findall(r"\([^)]*\)|\S+", coords)

253def strings_to_coords(
254    text: str | list[str],
255    when_noncoord: WhenMissing = "skip",
256) -> list[str | CoordTup]:
257    """converts a list of tokens to a list of coordinates
258
259    returns list[CoordTup] if `when_noncoord` is "skip" or "error"
260    returns list[str | CoordTup] if `when_noncoord` is "include"
261    """
262    warnings.warn(
263        "`util.strings_to_coords` only supports legacy UT strings. Function will be replaced with a generalized version in a future release.",
264        TokenizerPendingDeprecationWarning,
265    )
266    tokens_joined: str = text if isinstance(text, str) else " ".join(text)
267    tokens_processed: list[str] = coords_string_split_UT(tokens_joined)
268    result: list[str] = list()
269    for token in tokens_processed:
270        coord: CoordTup | None = coord_str_to_tuple_noneable(token)
271        if coord is None:
272            if when_noncoord == "skip":
273                continue
274            elif when_noncoord == "error":
275                raise ValueError(
276                    f"Invalid non-coordinate token '{token}' in text: '{text}'"
277                )
278            elif when_noncoord == "include":
279                result.append(token)
280            else:
281                raise ValueError(f"Invalid when_noncoord value '{when_noncoord}'")
282        else:
283            result.append(coord)
284    return result

287def coords_to_strings(
288    coords: list[str | CoordTup],
289    coord_to_strings_func: Callable[[CoordTup], list[str]],
290    when_noncoord: WhenMissing = "skip",
291) -> list[str]:
292    """converts a list of coordinates to a list of strings (tokens)
293
294    expects list[CoordTup] if `when_noncoord` is "error"
295    expects list[str | CoordTup] if `when_noncoord` is "include" or "skip"
296    """
297    result: list[str] = list()
298    for coord in coords:
299        if isinstance(coord, str):
300            if when_noncoord == "skip":
301                continue
302            elif when_noncoord == "error":
303                raise ValueError(
304                    f"Invalid non-coordinate '{coord}' in list of coords: '{coords}'"
305                )
306            elif when_noncoord == "include":
307                result.append(coord)
308            else:
309                raise ValueError(f"Invalid when_noncoord value '{when_noncoord}'")
310        else:
311            result.extend(coord_to_strings_func(coord))
312    return result

315def get_token_regions(toks: list[str]) -> tuple[list[str], list[str]]:
316    adj_list_start, adj_list_end = toks.index("<ADJLIST_START>") + 1, toks.index(
317        "<ADJLIST_END>"
318    )
319    adj_list = toks[adj_list_start:adj_list_end]
320    non_adj_list = toks[:adj_list_start] + toks[adj_list_end:]
321    return adj_list, non_adj_list

324def equal_except_adj_list_sequence(
325    rollout1: list[str],
326    rollout2: list[str],
327    do_except: bool = False,
328    when_counter_mismatch: ErrorMode = ErrorMode.EXCEPT,
329    when_len_mismatch: ErrorMode = ErrorMode.EXCEPT,
330) -> bool:
331    """Returns if the rollout strings are equal, allowing for differently sequenced adjacency lists.
332    <ADJLIST_START> and <ADJLIST_END> tokens must be in the rollouts.
333    Intended ONLY for determining if two tokenization schemes are the same for rollouts generated from the same maze.
334    This function should NOT be used to determine if two rollouts encode the same `LatticeMaze` object.
335
336    # Warning: CTT False Positives
337    This function is not robustly correct for some corner cases using `CoordTokenizers.CTT`.
338    If rollouts are passed for identical tokenizers processing two slightly different mazes, a false positive is possible.
339    More specifically, some cases of zero-sum adding and removing of connections in a maze within square regions along the diagonal will produce a false positive.
340    """
341
342    if len(rollout1) != len(rollout2):
343        if do_except:
344            when_len_mismatch.process(
345                f"Rollouts are not the same length: {len(rollout1)} != {len(rollout2)}"
346            )
347        return False
348    if ("<ADJLIST_START>" in rollout1) ^ ("<ADJLIST_START>" in rollout2):
349        if do_except:
350            raise ValueError(
351                f"Rollouts do not have the same <ADJLIST_START> token: `{'<ADJLIST_START>' in rollout1 = }` != `{'<ADJLIST_START>' in rollout2 = }`"
352            )
353        return False
354    if ("<ADJLIST_END>" in rollout1) ^ ("<ADJLIST_END>" in rollout2):
355        if do_except:
356            raise ValueError(
357                f"Rollouts do not have the same <ADJLIST_END> token: `{'<ADJLIST_END>' in rollout1 = }` != `{'<ADJLIST_END>' in rollout2 = }`"
358            )
359        return False
360
361    adj_list1, non_adj_list1 = get_token_regions(rollout1)
362    adj_list2, non_adj_list2 = get_token_regions(rollout2)
363    if non_adj_list1 != non_adj_list2:
364        if do_except:
365            when_len_mismatch.process(
366                f"Non-adjacency list tokens are not the same:\n{non_adj_list1}\n!=\n{non_adj_list2}"
367            )
368            raise ValueError(
369                f"Non-adjacency list tokens are not the same:\n{non_adj_list1}\n!=\n{non_adj_list2}"
370            )
371        return False
372    counter1: Counter = Counter(adj_list1)
373    counter2: Counter = Counter(adj_list2)
374    counters_eq: bool = counter1 == counter2
375    if not counters_eq:
376        if do_except:
377            when_counter_mismatch.process(
378                f"Adjacency list counters are not the same:\n{counter1}\n!=\n{counter2}\n{counter1 - counter2 = }"
379            )
380        return False
381
382    return True

385def connection_list_to_adj_list(
386    conn_list: ConnectionList,
387    shuffle_d0: bool = True,
388    shuffle_d1: bool = True,
389) -> Int8[np.ndarray, "conn start_end=2 coord=2"]:
390    """converts a `ConnectionList` (special lattice format) to a shuffled adjacency list
391
392    # Parameters:
393    - `conn_list: ConnectionList`
394        special internal format for graphs which are subgraphs of a lattice
395    - `shuffle_d0: bool`
396        shuffle the adjacency list along the 0th axis (order of pairs)
397    - `shuffle_d1: bool`
398        shuffle the adjacency list along the 1st axis (order of coordinates in each pair).
399        If `False`, all pairs have the smaller coord first.
400
401
402    # Returns:
403     - `Int8[np.ndarray, "conn start_end=2 coord=2"]`
404        adjacency list in the shape `(n_connections, 2, 2)`
405    """
406
407    n_connections: int = conn_list.sum()
408    adj_list: Int8[np.ndarray, "conn start_end=2 coord=2"] = np.full(
409        (n_connections, 2, 2), -1, dtype=np.int8
410    )
411
412    if shuffle_d1:
413        flip_d1: Float[np.ndarray, "conn"] = np.random.rand(n_connections)
414
415    # loop over all nonzero elements of the connection list
416    i: int = 0
417    for d, x, y in np.ndindex(conn_list.shape):
418        if conn_list[d, x, y]:
419            c_start: CoordTup = (x, y)
420            c_end: CoordTup = (
421                x + (1 if d == 0 else 0),
422                y + (1 if d == 1 else 0),
423            )
424            adj_list[i, 0] = np.array(c_start, dtype=np.int8)
425            adj_list[i, 1] = np.array(c_end, dtype=np.int8)
426
427            # flip if shuffling
428            if shuffle_d1 and (flip_d1[i] > 0.5):
429                c_s, c_e = adj_list[i, 0].copy(), adj_list[i, 1].copy()
430                adj_list[i, 0] = c_e
431                adj_list[i, 1] = c_s
432
433            i += 1
434
435    if shuffle_d0:
436        np.random.shuffle(adj_list)
437
438    return adj_list

441def is_connection(
442    edges: ConnectionArray, connection_list: ConnectionList
443) -> Bool[np.ndarray, "is_connection=edges"]:
444    """
445    Returns if each edge in `edges` is a connection (`True`) or wall (`False`) in `connection_list`.
446    """
447    sorted_edges = np.sort(edges, axis=1)
448    edge_direction = (
449        (sorted_edges[:, 1, :] - sorted_edges[:, 0, :])[:, 0] == 0
450    ).astype(np.int8)
451    return connection_list[edge_direction, sorted_edges[:, 0, 0], sorted_edges[:, 0, 1]]

130      - `coords[0]`: The previous location, used to determine the current absolute direction that the "agent" is facing.
131      - `coords[1]`: The current location
132      - `coords[2]`: The next location. May be equal to the current location.
133    """
134    if coords.shape != (3, 2):
135        raise ValueError(f"`coords` must have shape (3,2). Got {coords.shape} instead.")
136    directions = coords[1:] - coords[:-1]
137    if not np.all(np.linalg.norm(directions, axis=1) <= np.array([1.1, 1.1])):
138        # Use floats as constant since `np.linalg.norm` returns float array
139        raise ValueError(
140            f"Adjacent `coords` must be neighboring or equivalent. Got {coords} instead."
141        )
142    if np.array_equal(coords[1], coords[2]):
143        return VOCAB.PATH_STAY
144    if np.array_equal(coords[0], coords[2]):
145        return VOCAB.PATH_BACKWARD
146    if np.array_equal(coords[0], coords[1]):
147        raise ValueError(
148            f"Previous first-person direction indeterminate from {coords=}."
149        )
150    if np.array_equal(directions[0], directions[1]):
151        return VOCAB.PATH_FORWARD
152    directions = np.append(
153        directions, [[0], [0]], axis=1
154    )  # Augment to represent unit basis vectors in 3D
155    match np.cross(directions[0], directions[1])[-1]:
156        case 1:
157            return VOCAB.PATH_LEFT
158        case -1:
159            return VOCAB.PATH_RIGHT

389    shuffle_d1: bool = True,

391    """converts a `ConnectionList` (special lattice format) to a shuffled adjacency list
392
393    # Parameters:
394    - `conn_list: ConnectionList`
395        special internal format for graphs which are subgraphs of a lattice
396    - `shuffle_d0: bool`
397        shuffle the adjacency list along the 0th axis (order of pairs)
398    - `shuffle_d1: bool`
399        shuffle the adjacency list along the 1st axis (order of coordinates in each pair).
400        If `False`, all pairs have the smaller coord first.
401
402
403    # Returns:
404     - `Int8[np.ndarray, "conn start_end=2 coord=2"]`
405        adjacency list in the shape `(n_connections, 2, 2)`
406    """
407
408    n_connections: int = conn_list.sum()
409    adj_list: Int8[np.ndarray, "conn start_end=2 coord=2"] = np.full(
410        (n_connections, 2, 2), -1, dtype=np.int8
411    )
412
413    if shuffle_d1:
414        flip_d1: Float[np.ndarray, "conn"] = np.random.rand(n_connections)
415
416    # loop over all nonzero elements of the connection list
417    i: int = 0
418    for d, x, y in np.ndindex(conn_list.shape):
419        if conn_list[d, x, y]:
420            c_start: CoordTup = (x, y)
421            c_end: CoordTup = (
422                x + (1 if d == 0 else 0),
423                y + (1 if d == 1 else 0),
424            )
425            adj_list[i, 0] = np.array(c_start, dtype=np.int8)
426            adj_list[i, 1] = np.array(c_end, dtype=np.int8)
427
428            # flip if shuffling
429            if shuffle_d1 and (flip_d1[i] > 0.5):
430                c_s, c_e = adj_list[i, 0].copy(), adj_list[i, 1].copy()
431                adj_list[i, 0] = c_e
432                adj_list[i, 1] = c_s
433
434            i += 1
435
436    if shuffle_d0:
437        np.random.shuffle(adj_list)
438
