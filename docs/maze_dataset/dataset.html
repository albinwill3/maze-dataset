456    def _serialize_minimal(self) -> JSONitem:
457        "alternate serialization where metadata is collected and mazes are stored in concatenated form"
458        if self.generation_metadata_collected is None:
459            filtered_meta = self.filter_by.collect_generation_meta()
460        else:
461            filtered_meta = self
462
463        max_solution_len: int = max(m.solution.shape[0] for m in filtered_meta.mazes)
464        n_mazes: int = len(filtered_meta.mazes)
465        grid_n: int = filtered_meta.cfg.grid_n
466
467        maze_connection_lists: np.ndarray = np.empty(
468            (n_mazes, 2, grid_n, grid_n), dtype=np.bool_
469        )
470        # maze_endpoints: np.ndarray = np.empty((n_mazes, 2, 2), dtype=np.int8)
471        maze_solution_lengths: np.ndarray = np.empty((n_mazes,), dtype=np.int32)
472        maze_solutions: np.ndarray = np.empty(
473            (n_mazes, max_solution_len, 2), dtype=np.int8
474        )
475
476        for idx, maze in enumerate(filtered_meta.mazes):
477            maze_connection_lists[idx] = maze.connection_list
478            # maze_endpoints[idx] = np.array([maze.start_pos, maze.end_pos])
479            maze_solution_lengths[idx] = maze.solution.shape[0]
480            maze_solutions[idx, : maze.solution.shape[0]] = maze.solution
481
482        return dict(
483            __format__="MazeDataset:minimal",
484            cfg=json_serialize(filtered_meta.cfg),
485            generation_metadata_collected=json_serialize(
486                filtered_meta.generation_metadata_collected
487            ),
488            maze_connection_lists=maze_connection_lists,
489            # maze_endpoints=maze_endpoints,
490            maze_solution_lengths=maze_solution_lengths,
491            maze_solutions=maze_solutions,
492        )

494    def _serialize_minimal_soln_cat(self) -> JSONitem:
495        "alternate serialization where metadata is collected, and mazes and their solutions are stored in concatenated form"
496        if self.generation_metadata_collected is None:
497            filtered_meta = self.filter_by.collect_generation_meta()
498        else:
499            filtered_meta = self
500
501        maze_solution_lengths: np.ndarray = np.array(
502            [m.solution.shape[0] for m in filtered_meta.mazes],
503            dtype=np.int32,
504        )
505        n_mazes: int = len(filtered_meta.mazes)
506        grid_n: int = filtered_meta.cfg.grid_n
507        total_solution_len: int = np.sum(maze_solution_lengths)
508
509        maze_connection_lists: np.ndarray = np.empty(
510            (n_mazes, 2, grid_n, grid_n), dtype=np.bool_
511        )
512        maze_endpoints: np.ndarray = np.empty((n_mazes, 2, 2), dtype=np.int8)
513        maze_solutions_concat: np.ndarray = np.empty(
514            (total_solution_len, 2), dtype=np.int8
515        )
516
517        solutions_running_idx: int = 0
518        for idx, maze in enumerate(filtered_meta.mazes):
519            maze_connection_lists[idx] = maze.connection_list
520            maze_endpoints[idx] = np.array([maze.start_pos, maze.end_pos])
521            soln_len: int = maze.solution.shape[0]
522            maze_solution_lengths[idx] = soln_len
523            maze_solutions_concat[
524                solutions_running_idx : solutions_running_idx + soln_len
525            ] = maze.solution
526            solutions_running_idx += soln_len
527
528        return dict(
529            __format__="MazeDataset:minimal_soln_cat",
530            cfg=json_serialize(filtered_meta.cfg),
531            generation_metadata_collected=json_serialize(
532                filtered_meta.generation_metadata_collected
533            ),
534            maze_connection_lists=maze_connection_lists,
535            maze_endpoints=maze_endpoints,
536            maze_solution_lengths=maze_solution_lengths,
537            maze_solutions_concat=maze_solutions_concat,
538        )

 79        1. Choose the initial cell, mark it as visited and push it to the stack

 83                        1. Push the current cell to the stack
 84                        2. Choose one of the unvisited neighbours
 85                        3. Remove the wall between the current cell and the chosen cell
 86                        4. Mark the chosen cell as visited and push it to the stack
