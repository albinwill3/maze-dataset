246    def as_tokens(
247        self,
248        maze_tokenizer,  # TODO: MazeTokenizer
249        limit: int | None = None,
250        join_tokens_individual_maze: bool = False,
251    ) -> list[list[str]] | list[str]:
252        """return the dataset as tokens according to the passed `maze_tokenizer`
253
254        the `maze_tokenizer` should be either a `MazeTokenizer` or a `MazeTokenizerModular`
255
256        if `join_tokens_individual_maze` is True, then the tokens of each maze are
257        joined with a space, and the result is a list of strings.
258        i.e.:
259
260            >>> dataset.as_tokens(join_tokens_individual_maze=False)
261            [["a", "b", "c"], ["d", "e", "f"]]
262            >>> dataset.as_tokens(join_tokens_individual_maze=True)
263            ["a b c", "d e f"]
264        """
265        output: list[list[str]] = [
266            maze.as_tokens(maze_tokenizer) for maze in self.mazes[:limit]
267        ]
268        if join_tokens_individual_maze:
269            return [" ".join(tokens) for tokens in output]
270        else:
271            return output

273    def __len__(self) -> int:
274        return len(self.mazes)

277        if not isinstance(other, MazeDataset):
278            return NotImplemented
279        # TODO: compare hashes of data instead of the data itself?

282    @classmethod
283    def generate(
284        cls,
285        cfg: MazeDatasetConfig,
286        gen_parallel: bool = False,
287        pool_kwargs: dict | None = None,
288        verbose: bool = False,
289    ) -> "MazeDataset":
290        """generate a maze dataset given a config and some generation parameters"""
291
292        # avoid copying since we dont want to pickle the staticmethod, just load/serialize it to avoid modifying the original config
293        cfg_cpy = MazeDatasetConfig.load(cfg.serialize())
294
295        if pool_kwargs is None:
296            pool_kwargs = dict()
297        mazes: list[SolvedMaze] = list()
298        maze_indexes: Int[np.int8, "maze_index"] = np.arange(cfg_cpy.n_mazes)
299
300        solved_mazes: list[SolvedMaze]
301        tqdm_kwargs: dict = dict(
302            total=cfg_cpy.n_mazes,
303            unit="maze",
304            desc="generating & solving mazes",
305            disable=not verbose,
306        )
307        # TODO: don't use the global unless generating in parallel!
308        if gen_parallel:
309            with multiprocessing.Pool(
310                **pool_kwargs,
311                initializer=_maze_gen_init_worker,
312                initargs=(cfg_cpy,),
313            ) as pool:
314                solved_mazes = list(
315                    tqdm.tqdm(
316                        pool.imap(
317                            _generate_maze_helper,
318                            maze_indexes,
319                        ),
320                        **tqdm_kwargs,
321                    )
322                )
323        else:
324            _maze_gen_init_worker(cfg_cpy)
325            solved_mazes = list(
326                tqdm.tqdm(
327                    map(
328                        _generate_maze_helper,
329                        maze_indexes,
330                    ),
331                    **tqdm_kwargs,
332                )
333            )
334        # reset seed to default value
335        np.random.seed(cfg_cpy.seed)
336
337        return cls(
338            cfg=cfg_cpy,
339            mazes=solved_mazes,
340        )

346    @classmethod
347    def load(cls, data: JSONitem) -> "MazeDataset":
348        """load from zanj/json"""
349        if data["__format__"] == "MazeDataset:minimal":
350            return cls._load_minimal(data)
351        elif data["__format__"] == "MazeDataset:minimal_soln_cat":
352            return cls._load_minimal_soln_cat(data)
353        elif data["__format__"] == "MazeDataset":
354            if (
355                SERIALIZE_MINIMAL_THRESHOLD == -1
356            ):  # Allow access to `_load_legacy` for profiling
357                return cls._load_legacy(data)
358            return cls._load_full(data)
359        else:
360            raise KeyError(
361                f"`__format__` string {data['__format__']} is not a recognized `MazeDataset` format."
362            )

455    def _serialize_minimal(self) -> JSONitem:
456        "alternate serialization where metadata is collected and mazes are stored in concatenated form"
457        if self.generation_metadata_collected is None:
458            filtered_meta = self.filter_by.collect_generation_meta()
459        else:
460            filtered_meta = self
461
462        max_solution_len: int = max(m.solution.shape[0] for m in filtered_meta.mazes)
463        n_mazes: int = len(filtered_meta.mazes)
464        grid_n: int = filtered_meta.cfg.grid_n
465
466        maze_connection_lists: np.ndarray = np.empty(
467            (n_mazes, 2, grid_n, grid_n), dtype=np.bool_
468        )
469        # maze_endpoints: np.ndarray = np.empty((n_mazes, 2, 2), dtype=np.int8)
470        maze_solution_lengths: np.ndarray = np.empty((n_mazes,), dtype=np.int32)
471        maze_solutions: np.ndarray = np.empty(
472            (n_mazes, max_solution_len, 2), dtype=np.int8
473        )
474
475        for idx, maze in enumerate(filtered_meta.mazes):
476            maze_connection_lists[idx] = maze.connection_list
477            # maze_endpoints[idx] = np.array([maze.start_pos, maze.end_pos])
478            maze_solution_lengths[idx] = maze.solution.shape[0]
479            maze_solutions[idx, : maze.solution.shape[0]] = maze.solution
480
481        return dict(
482            __format__="MazeDataset:minimal",
483            cfg=json_serialize(filtered_meta.cfg),
484            generation_metadata_collected=json_serialize(
485                filtered_meta.generation_metadata_collected
486            ),
487            maze_connection_lists=maze_connection_lists,
488            # maze_endpoints=maze_endpoints,
489            maze_solution_lengths=maze_solution_lengths,
490            maze_solutions=maze_solutions,
491        )

493    def _serialize_minimal_soln_cat(self) -> JSONitem:
494        "alternate serialization where metadata is collected, and mazes and their solutions are stored in concatenated form"
495        if self.generation_metadata_collected is None:
496            filtered_meta = self.filter_by.collect_generation_meta()
497        else:
498            filtered_meta = self
499
500        maze_solution_lengths: np.ndarray = np.array(
501            [m.solution.shape[0] for m in filtered_meta.mazes],
502            dtype=np.int32,
503        )
504        n_mazes: int = len(filtered_meta.mazes)
505        grid_n: int = filtered_meta.cfg.grid_n
506        total_solution_len: int = np.sum(maze_solution_lengths)
507
508        maze_connection_lists: np.ndarray = np.empty(
509            (n_mazes, 2, grid_n, grid_n), dtype=np.bool_
510        )
511        maze_endpoints: np.ndarray = np.empty((n_mazes, 2, 2), dtype=np.int8)
512        maze_solutions_concat: np.ndarray = np.empty(
513            (total_solution_len, 2), dtype=np.int8
514        )
515
516        solutions_running_idx: int = 0
517        for idx, maze in enumerate(filtered_meta.mazes):
518            maze_connection_lists[idx] = maze.connection_list
519            maze_endpoints[idx] = np.array([maze.start_pos, maze.end_pos])
520            soln_len: int = maze.solution.shape[0]
521            maze_solution_lengths[idx] = soln_len
522            maze_solutions_concat[
523                solutions_running_idx : solutions_running_idx + soln_len
524            ] = maze.solution
525            solutions_running_idx += soln_len
526
527        return dict(
528            __format__="MazeDataset:minimal_soln_cat",
529            cfg=json_serialize(filtered_meta.cfg),
530            generation_metadata_collected=json_serialize(
531                filtered_meta.generation_metadata_collected
532            ),
533            maze_connection_lists=maze_connection_lists,
534            maze_endpoints=maze_endpoints,
535            maze_solution_lengths=maze_solution_lengths,
536            maze_solutions_concat=maze_solutions_concat,
537        )

580def register_maze_filter(
581    method: typing.Callable[[SolvedMaze, typing.Any], bool]
582) -> DatasetFilterProtocol:
583    """register a maze filter, casting it to operate over the whole list of mazes
584
585    method should be a staticmethod of a namespace class registered with `register_filter_namespace_for_dataset`
586
587    this is a more restricted version of `register_dataset_filter` that removes the need for boilerplate for operating over the arrays
588    """
589
590    @functools.wraps(method)
591    def wrapper(dataset: MazeDataset, *args, **kwargs):
