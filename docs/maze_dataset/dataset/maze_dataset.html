 223	def _to_ps_array(self) -> _PercolationSuccessArray:
 224		"""Convert this config to a [p, grid_n, deadends, endpoints_not_equal, generator_func] vector.
 225
 226		used in predicting the success rate
 227		"""
 228		try:
 229			assert self.maze_ctor.__name__ in _GENERATORS_PERCOLATED, (
 230				f"generator not supported, must be a percolation generator\n{self.maze_ctor.__name__ = }, {_GENERATORS_PERCOLATED = }"
 231			)
 232			assert "p" in self.maze_ctor_kwargs, (
 233				f"maze_ctor_kwargs must have a 'p' (percolation value) key: {self.maze_ctor_kwargs = }"
 234			)
 235			assert not self.endpoint_kwargs.get("except_on_no_valid_endpoint", True), (
 236				f"except_on_no_valid_endpoint must be False, or else if any maze fails to generate, the whole dataset will fail: {self.endpoint_kwargs = }"
 237			)
 238		except AssertionError as e:
 239			err_msg: str = f"invalid config for percolation success prediction: {self.summary() = }"
 240			raise NoPercolationInConfigError(
 241				err_msg,
 242			) from e
 243
 244		endpoints_unique_flag: int = int(
 245			self.endpoint_kwargs.get("endpoints_not_equal", True),
 246		)
 247
 248		# adjustment for bknutson0
 249		if not (
 250			self.endpoint_kwargs.get("deadend_start", False)
 251			and self.endpoint_kwargs.get("deadend_end", False)
 252		):
 253			# we didnt train on this, but if either endpoint is not required to be in a dead end
 254			# then  requiring the endpoints to be unique does not really affect the success rate
 255			# (except for very small percolation values, pure percolation generation)
 256			endpoints_unique_flag: int = 0
 257
 258		return np.array(
 259			[
 260				float(self.maze_ctor_kwargs["p"]),
 261				float(self.grid_n),
 262				float(
 263					int(
 264						self.endpoint_kwargs.get("deadend_start", False)
 265						or self.endpoint_kwargs.get("deadend_end", False),
 266					),
 267				),
 268				float(endpoints_unique_flag),
 269				float(_GENERATORS_PERCOLATED.index(self.maze_ctor.__name__)),
 270			],
 271			dtype=np.float64,
 272		)

 303	def success_fraction_estimate(
 304		self,
 305		except_if_all_success_expected: bool = False,
 306	) -> float:
 307		"""Estimate the success fraction of this config.
 308
 309		only valid when the generator is a percolation generator,
 310		and endpoints are enforced to be dead ends
 311
 312		this estimate comes from `estimate_dataset_fractions.ipynb` and `maze_dataset.benchmarks.sweep_fit`
 313
 314		# Parameters:
 315		- `except_if_all_success_expected : bool`
 316			if `True`, don't raise an error if the success fraction is below the threshold.
 317			will always return `1.0` if the config is not expected to fail
 318
 319		# Returns:
 320		- `float`
 321			estimated success fraction
 322
 323		# Raises:
 324		- `NoPercolationInConfigError` : if the config is not expected to fail, and `except_if_all_success_expected` is `False`
 325		"""
 326		try:
 327			return cfg_success_predict_fn(self)
 328
 329		except NoPercolationInConfigError as e:
 330			if except_if_all_success_expected:
 331				return 1.0
 332			else:
 333				raise e  # noqa: TRY201
 334
 335	def success_fraction_compensate(
 336		self,
 337		safety_margin: float = 1.2,
 338		except_if_all_success_expected: bool = False,
 339		epsilon: float = 1e-2,
 340	) -> "MazeDatasetConfig":
 341		"""return a new `MazeDatasetConfig` like this one with `n_mazes` adjusted to compensate for the success fraction
 342
 343		# Parameters:
 344		- `safety_margin : float`
 345			safety margin to apply to the success fraction estimate
 346			(defaults to `1.2`, or 20% more mazes than estimated)
 347		- `except_if_all_success_expected : bool`
 348			if `True`, don't raise an error if the success fraction is below the threshold.
 349			this is passed to `MazeDatasetConfig.success_fraction_estimate`.
 350			if your config isn't expected to fail, passing this might mean you generate more mazes than needed
 351			since `safety_margin` is still applied.
 352			(defaults to `False`)
 353		- `epsilon : float`
 354			raise `SuccessChanceTooSmallError` if the success fraction is below this threshold
 355			(defaults to `1e-2`)
 356
 357		# Returns:
 358		- `MazeDatasetConfig`
 359			new config with adjusted `n_mazes`
 360
 361		# Raises:
 362		- `SuccessChanceTooSmallError` : if the computed success fraction is below `epsilon`
 363		"""
 364		# compute and check the success fraction
 365		success_fraction: float = self.success_fraction_estimate(
 366			except_if_all_success_expected=except_if_all_success_expected,
 367		)
 368		if success_fraction < epsilon:
 369			err_msg: str = (
 370				f"{success_fraction = } is below the threshold of {epsilon = }"
 371			)
 372			raise SuccessChanceTooSmallError(
 373				err_msg,
 374			)
 375
 376		# compute the new number of mazes
 377		n_mazes: int = self.n_mazes
 378		new_n_mazes: int = int((n_mazes * safety_margin) / success_fraction) + 1
 379
 380		# put it in a new config and return
 381		cfg_dict: dict = self.serialize()
 382		cfg_dict["n_mazes"] = new_n_mazes
 383		return MazeDatasetConfig.load(cfg_dict)

 577		pool_kwargs: dict | None = None,
 578		verbose: bool = False,
 579	) -> "MazeDataset":
 580		"""Generate a maze dataset given a config and some generation parameters"""
 581		# Copy the config to avoid modifying the original
 582		cfg_cpy: MazeDatasetConfig = MazeDatasetConfig.load(
 583			json.loads(json.dumps(cfg.serialize())),
 584		)
 585
 586		if pool_kwargs is None:
 587			pool_kwargs = dict()
 588		maze_indexes: Int[np.ndarray, " maze_index"] = np.arange(cfg_cpy.n_mazes)  # type: ignore[assignment]
 589
 590		solved_mazes: list[SolvedMaze | None]
 591		# Configure tqdm for progress bar
 592		tqdm_kwargs: dict = dict(
 593			total=cfg_cpy.n_mazes,
 594			unit="maze",
 595			desc="generating & solving mazes",
 596			disable=not verbose,
 597		)
 598		# TODO: don't use the global unless generating in parallel!
 599		if gen_parallel:

 601				**pool_kwargs,
 602				initializer=_maze_gen_init_worker,
 603				initargs=(cfg_cpy,),
 604			) as pool:

 606					tqdm.tqdm(
 607						pool.imap(_generate_maze_helper, maze_indexes),
 608						**tqdm_kwargs,
 609					),
 610				)
 611
 612		else:
 613			_maze_gen_init_worker(cfg_cpy)
 614			solved_mazes = list(
 615				tqdm.tqdm(
 616					map(
 617						_generate_maze_helper,
 618						maze_indexes.tolist(),
 619					),
 620					**tqdm_kwargs,
 621				),
 622			)
 623
 624		# Filter out None values explicitly after ensuring all results are collected
 625		solved_mazes_: list[SolvedMaze] = [
 626			maze for maze in solved_mazes if maze is not None
 627		]
 628		# solved_mazes_ = list(filter(lambda x: x is not None, solved_mazes))

 631		cfg_cpy.n_mazes = len(solved_mazes_)

 642		return dataset
