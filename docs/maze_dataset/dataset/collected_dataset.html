 72    """a collection of maze datasets"""

 74    def __init__(
 75        self,
 76        cfg: MazeDatasetCollectionConfig,
 77        maze_datasets: list[MazeDataset],
 78        generation_metadata_collected: dict | None = None,
 79    ) -> None:
 80        super().__init__()
 81        self.cfg: MazeDatasetCollectionConfig = cfg
 82        self.maze_datasets: list[MazeDataset] = list(maze_datasets)
 83        for c, ds in zip(self.cfg.maze_dataset_configs, self.maze_datasets):
 84            assert c.name == ds.cfg.name
 85            assert c == ds.cfg
 86
 87        self.generation_metadata_collected: dict | None = generation_metadata_collected
 88
 89    @property
 90    def dataset_lengths(self) -> list[int]:
 91        return [len(dataset) for dataset in self.maze_datasets]
 92
 93    @property
 94    def dataset_cum_lengths(self) -> Int[np.ndarray, "indices"]:
 95        return np.array(list(itertools.accumulate(self.dataset_lengths)))
 96
 97    @cached_property
 98    def mazes(self) -> list[LatticeMaze]:
 99        return list(
100            itertools.chain.from_iterable(
101                dataset.mazes for dataset in self.maze_datasets
102            )
103        )
104
105    def __len__(self) -> int:
106        return sum(len(dataset) for dataset in self.maze_datasets)
107
108    def __getitem__(self, index: int):
109        # find which dataset the index belongs to
110        # we add 1, since np.searchsorted returns the
111        # index of the last element that is strictly less than the target
112        # while we want the index of the last element less than or equal to the target
113        dataset_idx: int = np.searchsorted(self.dataset_cum_lengths, index + 1)
114        index_adjusted: int = index
115        if dataset_idx > 0:
116            # if the index is 0, `dataset_idx - 1` will be -1.
117            # We just want to use the base index
118            index_adjusted -= self.dataset_cum_lengths[dataset_idx - 1]
119        return self.maze_datasets[dataset_idx][index_adjusted]
120

122    def generate(
123        cls, cfg: MazeDatasetCollectionConfig, **kwargs
124    ) -> "MazeDatasetCollection":
125        datasets = [
126            MazeDataset.generate(config, **kwargs)
127            for config in cfg.maze_dataset_configs
128        ]
129        return cls(cfg, datasets)

158            }

161    # TODO: remove duplication with MazeDatasetConfig().as_tokens() somehow?
162    def as_tokens(
163        self,
164        maze_tokenizer,  # TODO: MazeTokenizer
165        limit: int | None = None,
166        join_tokens_individual_maze: bool = False,
167    ) -> list[list[str]] | list[str]:
168        """return the dataset as tokens
169
170        if join_tokens_individual_maze is True, then the tokens of each maze are
171        joined with a space, and the result is a list of strings.
172        i.e.:
173        >>> dataset.as_tokens(join_tokens_individual_maze=False)
174        [["a", "b", "c"], ["d", "e", "f"]]
175        >>> dataset.as_tokens(join_tokens_individual_maze=True)
176        ["a b c", "d e f"]
177        """
178        output: list[list[str]] = [
179            maze.as_tokens(maze_tokenizer) for maze in self.mazes[:limit]
180        ]
181        if join_tokens_individual_maze:
182            return [" ".join(tokens) for tokens in output]
183        else:
184            return output

188        self.cfg.__dict__["n_mazes"] = len(self)
189        for dataset in self.maze_datasets:
190            dataset.update_self_config()
