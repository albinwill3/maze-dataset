 33def _extend_pixels(
 34    image: Int[np.ndarray, "x y rgb"], n_mult: int = 2, n_bdry: int = 1
 35) -> Int[np.ndarray, "n_mult*x+2*n_bdry n_mult*y+2*n_bdry rgb"]:
 36    wall_fill: int = PixelColors.WALL[0]
 37    assert all(
 38        x == wall_fill for x in PixelColors.WALL
 39    ), "PixelColors.WALL must be a single value"
 40
 41    output: np.ndarray = np.repeat(
 42        np.repeat(
 43            image,
 44            n_mult,
 45            axis=0,
 46        ),
 47        n_mult,
 48        axis=1,
 49    )
 50
 51    # pad on all sides by n_bdry
 52    output = np.pad(
 53        output,
 54        pad_width=((n_bdry, n_bdry), (n_bdry, n_bdry), (0, 0)),
 55        mode="constant",
 56        constant_values=wall_fill,
 57    )
 58
 59    return output

 69def process_maze_rasterized_input_target(
 70    maze: SolvedMaze,
 71    remove_isolated_cells: bool = True,
 72    extend_pixels: bool = True,
 73    endpoints_as_open: bool = False,
 74) -> Float[torch.Tensor, "in/tgt=2 x y rgb=3"]:
 75    # problem and solution mazes
 76    maze_pixels: PixelGrid = maze.as_pixels(show_endpoints=True, show_solution=True)
 77    problem_maze: PixelGrid = maze_pixels.copy()
 78    solution_maze: PixelGrid = maze_pixels.copy()
 79
 80    # in problem maze, set path to open
 81    problem_maze[(problem_maze == PixelColors.PATH).all(axis=-1)] = PixelColors.OPEN
 82
 83    # wherever solution maze is PixelColors.OPEN, set it to PixelColors.WALL
 84    solution_maze[(solution_maze == PixelColors.OPEN).all(axis=-1)] = PixelColors.WALL
 85    # wherever it is solution, set it to PixelColors.OPEN
 86    solution_maze[(solution_maze == PixelColors.PATH).all(axis=-1)] = PixelColors.OPEN
 87    if endpoints_as_open:
 88        for color in (PixelColors.START, PixelColors.END):
 89            solution_maze[(solution_maze == color).all(axis=-1)] = PixelColors.OPEN
 90
 91    # postprocess to match original easy_2_hard dataset
 92    if remove_isolated_cells:
 93        problem_maze = _remove_isolated_cells(problem_maze)
 94        solution_maze = _remove_isolated_cells(solution_maze)
 95
 96    if extend_pixels:
 97        problem_maze = _extend_pixels(problem_maze)
 98        solution_maze = _extend_pixels(solution_maze)
 99
100    return torch.tensor(np.array([problem_maze, solution_maze]))

208def make_numpy_collection(
209    base_cfg: RasterizedMazeDatasetConfig,
210    grid_sizes: list[int],
211    from_config_kwargs: dict | None = None,
212    verbose: bool = True,
213    key_fmt: str = "{size}x{size}",
214) -> dict[
215    typing.Literal["configs", "arrays"],
216    dict[str, RasterizedMazeDatasetConfig | np.ndarray],
217]:
218    """create a collection of configs and arrays for different grid sizes, in plain tensor form
219
220    output is of structure:
221    ```
222    {
223        "configs": {
224            "<n>x<n>": RasterizedMazeDatasetConfig,
225            ...
226        },
227        "arrays": {
228            "<n>x<n>": np.ndarray,
229            ...
230        },
231    }
232    ```
233    """
234
235    if from_config_kwargs is None:
236        from_config_kwargs = {}
237
238    datasets: dict[int, RasterizedMazeDataset] = {}
239
240    for size in grid_sizes:
241        if verbose:
242            print(f"Generating dataset for maze size {size}...")
243
244        cfg_temp: RasterizedMazeDatasetConfig = RasterizedMazeDatasetConfig.load(
245            base_cfg.serialize()
246        )
247        cfg_temp.grid_n = size
248
249        datasets[size] = RasterizedMazeDataset.from_config_augmented(
250            cfg=cfg_temp,
251            **from_config_kwargs,
252        )
253
254    return dict(
255        configs={
256            key_fmt.format(size=size): dataset.cfg for size, dataset in datasets.items()
257        },
258        arrays={
259            # get_batch(None) returns a single tensor of shape (n, 2, x, y, 3)
260            key_fmt.format(size=size): dataset.get_batch(None).cpu().numpy()
261            for size, dataset in datasets.items()
262        },
263    )
